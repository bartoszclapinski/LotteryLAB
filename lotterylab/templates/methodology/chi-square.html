from scipy.stats import chi2

def chi_square_test(observed: dict) -> dict:
    """{{ i18n.chi_code_docstring }}"""
    
    total = sum(observed.values())
    expected = total / len(observed)  # {{ i18n.chi_code_comment1 }}
    
    chi_sq = 0
    for obs in observed.values():
        chi_sq += (obs - expected) ** 2 / expected
    
    df = len(observed) - 1  # {{ i18n.chi_code_comment2 }}
    p_value = 1 - chi2.cdf(chi_sq, df)
    
    return {
        "chi_square": chi_sq,
        "p_value": p_value,
        "is_random": p_value > 0.05
    }
