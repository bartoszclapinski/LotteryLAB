import math
from collections import Counter


def shannon_entropy(numbers: list[int]) -> dict:
    """{{ i18n.entropy_code_docstring }}"""

    # {{ i18n.entropy_code_comment1 }}
    counts = Counter(numbers)
    total = sum(counts.values())
    if total == 0:
        return {'entropy': 0.0, 'max_entropy': 0.0, 'normalized': 0.0}

    # {{ i18n.entropy_code_comment2 }}
    probs = [count / total for count in counts.values()]

    # {{ i18n.entropy_code_comment3 }}
    H = -sum(p * math.log2(p) for p in probs if p > 0)
    max_entropy = math.log2(49)
    normalized = H / max_entropy if max_entropy > 0 else 0.0

    return {
        'entropy': H,
        'max_entropy': max_entropy,
        'normalized': normalized,
    }

numbers = [5, 12, 23, 8, 31, 17, ...]
result = shannon_entropy(numbers)

print("{{ i18n.entropy_code_print2 }}:", result['normalized'])
