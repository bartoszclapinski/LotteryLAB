import numpy as np
from scipy import stats

def runs_test(sequence: list) -> dict:
    """{{ i18n.runs_code_docstring }}"""
    
    # {{ i18n.runs_code_comment1 }}
    median = np.median(sequence)
    binary = [1 if x >= median else 0 for x in sequence]
    
    # {{ i18n.runs_code_comment2 }}
    runs = 1
    for i in range(1, len(binary)):
        if binary[i] != binary[i-1]:
            runs += 1
    
    # {{ i18n.runs_code_comment3 }}
    n1 = sum(binary)
    n2 = len(binary) - n1
    
    # {{ i18n.runs_code_comment4 }}
    mu = (2 * n1 * n2) / (n1 + n2) + 1
    sigma = np.sqrt((2*n1*n2*(2*n1*n2-n1-n2)) / 
                    ((n1+n2)**2 * (n1+n2-1)))
    
    # {{ i18n.runs_code_comment5 }}
    z = (runs - mu) / sigma
    p_value = 2 * (1 - stats.norm.cdf(abs(z)))
    
    return {
        "runs": runs,
        "expected": mu,
        "z_score": z,
        "p_value": p_value,
        "is_random": p_value > 0.05
    }
