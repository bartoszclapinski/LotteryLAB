import numpy as np
from scipy.stats import chi2

def monte_carlo_lottery(n_simulations: int, draws_per_sim: int, real_chi: float) -> dict:
    """{{ i18n.monte_code_docstring }}"""
    
    chi_values = []
    
    for _ in range(n_simulations):
        # {{ i18n.monte_code_comment1 }}
        draws = np.random.randint(1, 50, size=(draws_per_sim, 6))
        
        # {{ i18n.monte_code_comment2 }}
        counts = np.bincount(draws.flatten(), minlength=50)[1:]
        expected = draws_per_sim * 6 / 49
        chi_sq = np.sum((counts - expected) ** 2 / expected)
        chi_values.append(chi_sq)
    
    chi_values = np.array(chi_values)
    
    # {{ i18n.monte_code_comment3 }}
    percentile = np.sum(chi_values <= real_chi) / n_simulations * 100
    
    return {
        'mean_chi': np.mean(chi_values),
        'std_chi': np.std(chi_values),
        'percentile': percentile,
        'in_95_ci': 2.5 <= percentile <= 97.5
    }
