from scipy import stats
import numpy as np

def ks_test_lottery(frequencies: dict) -> dict:
    """{{ i18n.ks_code_docstring }}"""
    
    # {{ i18n.ks_code_comment1 }}
    observed = np.array(list(frequencies.values()))
    total = observed.sum()
    
    # {{ i18n.ks_code_comment2 }}
    observed_norm = observed / total
    expected_norm = np.ones(len(observed)) / len(observed)
    
    # {{ i18n.ks_code_comment3 }}
    ecdf = np.cumsum(observed_norm)
    tcdf = np.cumsum(expected_norm)
    
    # {{ i18n.ks_code_comment4 }}
    d_stat = np.max(np.abs(ecdf - tcdf))
    
    # {{ i18n.ks_code_comment5 }}
    n = len(observed)
    p_value = 2 * np.exp(-2 * n * d_stat**2)
    
    return {
        "d_statistic": d_stat,
        "p_value": min(p_value, 1.0),
        "is_uniform": p_value > 0.05
    }
